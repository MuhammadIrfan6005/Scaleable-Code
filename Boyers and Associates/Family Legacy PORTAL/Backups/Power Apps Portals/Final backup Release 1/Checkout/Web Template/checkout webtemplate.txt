      function totalchildsprice(name) {
        //For gift price calculation
        var giftamount = 0;
        var givemonthlyis = localStorage.getItem("givemonthly");
        if (givemonthlyis == 'true') {
          giftamount = localStorage.getItem("givemonthlyprice");
